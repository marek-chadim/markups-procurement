\begin{frame}{Why Should You Care?}

\begin{itemize}
    \item Public procurement = \textbf{\$1.3 trillion/year} across OECD (12\% of GDP)
    \item Czech construction = \textbf{49.6\%} of all Czech procurement contracts
    \item Yet: surprisingly little evidence on how procurement affects \emph{firm pricing power}
\end{itemize}

\bigskip

\accenttext{This paper:} Do government contractors charge higher markups?

\bigskip

\begin{itemize}
    \item \textbf{Answer:} Yes --- procurement premium $\approx$ \highlighttext{15\%}, declining from 30\% (2006) to 10\% (2021)
    \item \textbf{But:} the markup \emph{level} depends critically on specification choices
    \item \textbf{Methodological insight:} variable input choice and Markov specification dominate all other modeling decisions --- including the estimator itself
\end{itemize}

\end{frame}

% ============================================================
% MARKUP FORMULA
% ============================================================
\begin{frame}{How Are Markups Measured?}

Start from cost minimization (De Loecker \& Warzynski, 2012):

\begin{equation*}
    \boxed{\mu_{it} = \frac{\theta^V_{it}}{\alpha^V_{it}}}
\end{equation*}

where:
\begin{itemize}
    \item $\theta^V_{it} = \partial \ln Y / \partial \ln V$ \quad (output elasticity --- from production function)
    \item $\alpha^V_{it} = P^V V / PY$ \quad (expenditure share --- from data)
\end{itemize}

\bigskip

\textbf{Key question:} what is the ``variable input'' $V$?

\medskip

\begin{itemize}
    \item \accenttext{COGS} (cost of goods sold, $\approx$50\% of GO) $\Rightarrow$ markups $\approx$ 1.7--2.2
    \item \highlighttext{Intermediate inputs} ($\approx$83\% of GO) $\Rightarrow$ markups $\approx$ 1.15
    \item This choice \emph{flips the sign} of the procurement premium
\end{itemize}

\end{frame}

% ============================================================
% ESTIMATION
% ============================================================
\begin{frame}{Estimation: ACF Control Function}

\textbf{Translog production function:}
\begin{equation*}
    y_{it} = \beta_k k_{it} + \beta_v v_{it} + \beta_{kk} k_{it}^2 + \beta_{vv} v_{it}^2 + \beta_{kv} k_{it} v_{it} + \omega_{it} + \varepsilon_{it}
\end{equation*}

\textbf{Two stages:}
\begin{enumerate}
    \item \textbf{First stage:} OLS with 3rd-order polynomial in $(k, v)$ interacted with procurement dummy + year FE $\Rightarrow$ predict $\hat{\phi}_{it}$
    \item \textbf{Second stage:} GMM --- minimize $E[\xi_{it} \cdot Z_{it}] = 0$ where $\xi_{it}$ is innovation to Markov process
    \begin{equation*}
        \omega_{it} = g(\omega_{it-1}, pp_{it-1}) + \xi_{it}
    \end{equation*}
\end{enumerate}

\medskip

\textbf{Instruments:} lagged capital $k_{it-1}$, lagged COGS $v_{it-1}$

\medskip

Standard errors: ACH (2012) sandwich, firm-clustered, delta method for markups

\end{frame}

\begin{frame}{Why Does the Variable Input Matter So Much?}

\textbf{The markup formula is $\mu = \theta / \alpha$.} As the input's share of revenue ($\alpha$) grows:

\medskip

\begin{center}
\begin{tabular}{lccc}
\toprule
Input & Share of GO & Median $\mu$ & 90/50 ratio \\
\midrule
Labor (wages) & $\sim$17\% & 4.79 & 1.44 \\
COGS & $\sim$50\% & 2.02 & 1.23 \\
Composite (W+COGS) & $\sim$67\% & 1.48 & 1.21 \\
\bottomrule
\end{tabular}
\end{center}

\medskip

\begin{itemize}
    \item Raval (2023): markups from different inputs are \emph{positively} correlated ($\hat{\beta} = 0.28$) but far from $\beta = 1$ $\Rightarrow$ Hicks neutrality rejected
    \item DLEU Reply (2025): output elasticities are \emph{constant} over time --- all the action is in $\alpha$
    \item CMOY (2023): DLEU-style markups don't predict industry prices ($R^2 < 0.001$)
\end{itemize}

\medskip

\highlighttext{Good news:} the \emph{ranking} of firms is preserved $\Rightarrow$ treatment effect is more robust than levels

\end{frame}

% ============================================================
% WHAT SHOULD RESEARCHERS DO
% ============================================================
\begin{frame}{What Should Researchers Do?}

\begin{enumerate}
    \item \textbf{Report multiple variable inputs.} COGS and II give qualitatively different answers. The Raval (2023) overidentification test helps diagnose the problem.

    \medskip

    \item \textbf{Include the treatment dummy in the Markov process.} Omitting it understates the premium by a factor of 3. This is a modeling choice, not a robustness check.

    \medskip

    \item \textbf{Don't obsess over the estimator.} ACF $\approx$ OLS (GNR). Functional form doesn't matter much either. The imperfect competition correction is a second-order concern.

    \medskip

    \item \textbf{Focus on the treatment effect, not the markup level.} Absolute markups are mechanical (driven by $\alpha$). The \emph{relative} premium is what's informative about market power.
\end{enumerate}

\end{frame}

\begin{frame}{Primary Identification: Full-Panel Fixed Effects}

\textbf{Firm FE + Year FE} on 1,267 firms (474 switchers):
$$\ln\mu_{it} = \highlighttext{0.131} \times pp_{it} + \alpha_i + \gamma_t + \nu_{it}$$

\pause
\medskip
\begin{itemize}
    \item Premium = \highlighttext{14.0\%} (SE = 0.005) --- \accenttext{6,849 observations}
    \item Stable across specs: pooled OLS (15.0\%), firm FE (14.8\%), + controls (14.1\%)
    \item By NACE: buildings 17.4\%, civil eng.\ 6.4\%, specialized 11.1\%
\end{itemize}

\pause
\medskip
\textbf{Oster (2019) bounds:} Identified set $[12.3\%, 15.0\%]$ \\
$\delta^* = -5.9$ --- unobservables would need to work in the \emph{opposite} direction \\
Even removing Baran\v{e}k \& Titl (2024) +6\% favoritism $\rightarrow$ residual \highlighttext{$\sim$9\%}

\end{frame}

\begin{frame}{DLW Treatment Evaluation: Entry and Exit}

Following De Loecker \& Warzynski (2012) exporter framework:

\medskip
\begin{itemize}
    \item Entry into procurement: \highlighttext{+11.5\%} markup increase
    \item Exit from procurement: \highlighttext{$-$11.2\%} markup decrease
    \item \accenttext{Symmetric reversal} = strongest causal evidence
\end{itemize}

\pause
\medskip
\textbf{Productivity decomposition:}
\begin{itemize}
    \item Translog $\omega$ explains \accenttext{76\%} of premium (DLW: $\sim$70\% for exports)
    \item Residual after $\omega$: 3.2\% ($t = 4.1$) --- \emph{pricing power beyond productivity}
    \item Intensive margin: +3.7\% per unit of procurement share (dose-response)
\end{itemize}

\end{frame}

\begin{frame}{Double Wage Markdown}

\textbf{KLMS equation (17):} $W_{jt} = \underbrace{(1+\theta)^{-1}}_{\text{markdown}} \times \underbrace{(1-\epsilon)}_{\text{inv.\ markup}} \times P_{jt} \cdot \text{MPL}_{jt}$

\pause
\medskip
\begin{center}
\begin{tabular}{lccc}
\toprule
& Markdown & Inv.\ markup & \textbf{Double markdown} \\
\midrule
Czech construction & 0.889 & 0.711 & \highlighttext{0.632} \\
\textcolor{muted}{KLMS US (DiD)} & \textcolor{muted}{0.803} & \textcolor{muted}{0.863} & \textcolor{muted}{0.693} \\
\bottomrule
\end{tabular}
\end{center}

\pause
\medskip
\begin{itemize}
    \item Wages are \highlighttext{37\% below} the value of MPL (KLMS US: 31--33\%)
    \item Czech product demand \emph{less} elastic ($1{-}\epsilon = 0.71$ vs 0.86) --- smaller, more concentrated markets
    \item Czech labor \emph{more} elastic ($1/\theta = 8.0$ vs 4.1) --- larger informal sector
    \item \accenttext{Attenuation:} product + labor power partially cancel $\rightarrow$ explains null ADL correction
\end{itemize}

\end{frame}

\begin{frame}{Political Connections: Baran\v{e}k \& Titl (2024, JLE)}

\textbf{Key confounder:} politically connected firms overprice by \highlighttext{+6\%}
\begin{itemize}
    \item Electoral variation: connection disappears when party loses power
    \item No quality gains --- pure rent extraction
    \item EU oversight \emph{eliminates} favoritism (intergovernment level)
\end{itemize}

\pause
\medskip
\textbf{Implication for our premium:}
\begin{itemize}
    \item 14\% premium $-$ 6\% favoritism = \accenttext{$\sim$8\% residual} (still large)
    \item Oster bounds: $\delta^* = -5.9$ (robust even under worst-case)
    \item DLW: $\omega$ explains 76\% $\rightarrow$ most of premium is productivity selection
    \item \accenttext{Remaining 3.2\%}: pricing power + favoritism + other mechanisms
\end{itemize}

\end{frame}

% CONCLUSION
% ============================================================
\begin{frame}{Summary}

\begin{enumerate}
    \item \highlighttext{14\% procurement markup premium} --- firm FE on 1,267 firms, Oster-robust, confirmed across 9 methods and 5 industries on independent Orbis data

    \medskip

    \item \textbf{Specification sensitivity} propagates to treatment effects --- variable input choice flips the sign, Markov specification triples it. \accenttext{Extends BMY (2026) from levels to ATT}

    \medskip

    \item \textbf{Double market power:} Czech construction $\theta = 0.13$ (Orbis), double markdown = 0.63 --- wages 37\% below VMPL. Attenuation explains null ADL correction

    \medskip

    \item The premium \textbf{declined 30\% $\rightarrow$ 10\%} (2006--2021); productivity explains 76\% (DLW), political connections 6\% (B\&T), residual 3.2\%
\end{enumerate}
